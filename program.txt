const arr = [124, 123, 3];

var bubbleSort = () -> {
    const size = 2;
    var i: int = 0;

    while (i < size)
    {
        var j: int = 0;
        while (j < size - i)
        {
            if (arr[j] > arr[j + 1])
            {
                const temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            };
            j = j + 1;
        };
        i = i + 1;
    };
};

bubbleSort();
print(arr[0]);

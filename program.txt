const arr = [124, 123, 3];

const size = 2;
const i = [0];

while (i[0] < size)
{
    const j = [0];
    while (j[0] < size - i[0])
    {
        if (arr[j[0]] > arr[j[0] + 1])
        {
            const temp = arr[j[0]];
            arr[j[0]] = arr[j[0] + 1];
            arr[j[0] + 1] = temp;
        };
        j[0] = j[0] + 1;
    };
    i[0] = i[0] + 1;
};

print(arr[0]);
